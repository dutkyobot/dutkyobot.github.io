Firebase URL Shortener

// Initialize Firebase const app = initializeApp(firebaseConfig); const db = getFirestore(app); const functions = getFunctions(app); // Shorten URL const shortenBtn = document.getElementById("shortenBtn"); shortenBtn.addEventListener("click", async () => { const url = document.getElementById("urlInput").value; const shortenUrlFunction = httpsCallable(functions, 'shortenUrl'); try { const result = await shortenUrlFunction({ originalUrl: url }); const shortUrl = result.data.shortUrl; document.getElementById("shortUrl").innerText = `Shortened URL: ${shortUrl}`; } catch (error) { console.error("Error shortening URL:", error); } });
